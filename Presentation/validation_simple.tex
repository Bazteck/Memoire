\begin{frame}{Modèle de Propagation - Log-Distance}
	\centering
	\textbf{Équation du Modèle}:
	$$\text{PL}(d) = \text{PL}(d_0) + 10n \log_{10}\left(\frac{d}{d_0}\right) + X_\sigma$$
	
	\vspace{0.5cm}
	\textbf{Paramètres du Modèle}:
	\begin{itemize}
		\item $n = 3.3$ (exposant de perte de propagation)
		\item $\text{PL}(d_0) = 125$ dB (perte à distance référence $d_0 = 1$ km)
		\item $\sigma = 7$ dB (écart-type du shadowing)
	\end{itemize}
\end{frame}
