	\section{Méthodologie d'évaluation DDQN}
	% ============================================================
	
	\begin{frame}{Protocole d'évaluation de l'agent DDQN}
		\framesubtitle{Standard contre Agent DDQN — conditions identiques}
		
		\begin{columns}[T]
			
			\column{0.48\textwidth}
			\begin{block}{Configuration Standard}
				\begin{itemize}\small
					\item DR fixe, puissance fixe à \textbf{14 dBm}
					\item 30 min · 10 nœuds · 13 distances
					\item Taille payload = 1 Byte
					\item Séparé pour chaque DR (DR8--DR11)
				\end{itemize}
			\end{block}
			
			\vspace{0.4em}
			\begin{block}{ Configuration DDQN}
				\begin{itemize}\small
					\item Sélection \textbf{dynamique} du DR et de la puissance
					\item Même positions, même seed, même nœuds, même shadowing  que Standard
					\item Même configuration ,nombre de nœuds, temps simulation, intervalle que standard.
				\end{itemize}
			\end{block}
			
			\column{0.48\textwidth}
			\begin{exampleblock}{ Métriques collectées}
				\begin{itemize}\small
					\item \textbf{Fiabilité :} PDR
					\item \textbf{Énergie :}
					$E_{\text{tx}}$ [mJ] par transmission
					\item \textbf{Adaptativité :} distribution des choix DR et puissance vs distance
				\end{itemize}
			\end{exampleblock}
		
		\end{columns}
	\end{frame}
	\begin{frame}{Définition de termes}
			\begin{block}{Définition}
			\begin{itemize}
				\item La simulation standard désigne la simulation avec des DR(8/9/10/11) fixe sur toute les distances et une puissance de 14dBm. 
				\item Les comparaisons globales de DDQN à la simulation \textbf{Standard}, sont des comparaisons des métriques de l'agent DDQN a la moyenne des  résultats obtenu individuellement avec chaque DR (8,9,10,11) et une puissance d’émission a 14 dBm sur chacune de distances (139m-4000m).
			\end{itemize}
		\end{block}
	\end{frame}

		\label{tab:debit_distance}
		\begin{tabular}{|c|c|c|c|c|}
			\hline
			\rowcolor{blue!50}
			\color{white} \textbf{Distance (m)} & \color{white} \textbf{DR8/10 (bps)} & \color{white} \textbf{DR9/11 (bps)} & \color{white}  \textbf{Moyenne (bps)} & \color{white}\textbf{DDQN (bps)} \\
			\hline
			139    & 89,59  & 142,05 & 115,82 & 142,05 \\
			\hline
			331    & 89,59  & 142,05 & 115,82 & 140,73 \\
			\hline
			775    & 89,59  & 142,05 & 115,82 & 90,11 \\
			\hline
			1015-4000 & 89,59  & 142,05 & 115,82  & 89,59 \\
			\hline
		\end{tabular}
	\end{table}
\begin{itemize}
	
	\item À courte distance (139 m), le DDQN atteint son \textbf{débit maximal} de 142,05 bps. Cela s’explique par l’utilisation exclusive du DR9/11 (100\%), qui permet un débit plus élevé.
	
	\item À 331 m, le DDQN reste proche du débit maximal (140,73 bps), grâce à une utilisation encore majoritaire du DR9/11(98\%) et DR8/10(2\%).
	
	\item À moyenne distance (775 m), le débit du DDQN chute à 90,11 bps. Cela correspond à une transition vers le DR8/10 (99\%) et DR9/11 (1\%), utilisé pour maintenir la fiabilité de la transmission à plus longue distance.
	
	\item Sur les longues distances (1015–4000 m), le DDQN utilise exclusivement le DR8/10 (100\%), ce qui explique le débit réduit (89,59 bps). Cette stratégie illustre la capacité de l’agent DDQN à adapter dynamiquement le DR en fonction de la distance pour maximiser la fiabilité au détriment d'un débit élevé.
	
	\item En comparaison, la \textbf{moyenne} entre DR8/10 et DR9/11 est de 115,82 bps, mais le DDQN peut dépasser cette moyenne à courte distance grâce à la sélection intelligente du DR le plus performant.
\end{itemize}
\end{frame}
